\section{Board Specifications}
\begin{description}
  \item[Topology:] Active Clamp Flyback
  \item[Input voltage range:] 90–264 V AC, 47–63 Hz (typ. 230 V AC)
  \item[Output voltages:] Selectable 5, 9, 12, 15, 20 V DC
  \item[Switching frequency:] Up to 500 kHz
  \item[Main power switches:] GaN transistors
  \item[Magnetic components:] Transformer
  \item[Main capacitors:] 
    \begin{itemize}
      \item Input: $\sim$120 µF
      \item Output: $2 \times 330\,\mu$F + $3 \times 22\,\mu$F
    \end{itemize}
  \item[clamp networks:] active clamping
  \item[SR Controller:] SRK1004
   
\end{description}
